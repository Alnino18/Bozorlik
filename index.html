function doGet(e) {
  try {
    var raw = e.parameter.data;
    var data = JSON.parse(decodeURIComponent(raw));
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    if (data.type === "purchase") {
      var sh = ss.getSheetByName("Закупка")
             || ss.insertSheet("Закупка");
      if (sh.getLastRow() === 0)
        sh.appendRow(["Сана","Вақт","Бозор",
          "Касса","Харажат","Қолдиқ","Товарлар","Изоҳ"]);
      sh.appendRow([data.date, data.time, data.market,
        data.cashBalance, data.cashTotal, data.balance,
        data.items.map(function(i){
          return i.name+(i.kg?" "+i.kg+"кг":"")+" - "+i.totalPrice+" сўм";
        }).join("; "), data.note || ""]);
    }

    if (data.type === "debt") {
      var sh2 = ss.getSheetByName("Қарзлар")
              || ss.insertSheet("Қарзлар");
      if (sh2.getLastRow() === 0)
        sh2.appendRow(["Сана","Ким","Бозор",
          "Товар","кг","Нарх","Жами","Сана2"]);
      sh2.appendRow([data.date, data.who, data.market,
        data.name, data.kg, data.unitPrice,
        data.totalPrice, data.savedAt]);
    }

    if (data.type === "full_backup") {
      var shb = ss.getSheetByName("Backup")
              || ss.insertSheet("Backup");
      if (shb.getLastRow() === 0)
        shb.appendRow(["Backup Сана","Касса","Тарих сони",
          "Қарз сони","Жами қарз","Қарзлар"]);
      var debtNames = (data.debts||[]).map(function(d){
        return d.name+" "+d.totalPrice+" сўм"+(d.who?" ("+d.who+")":"");
      }).join("; ");
      shb.appendRow([data.backupDate, data.cashBalance,
        data.historyCount, data.debtCount,
        data.debtTotal, debtNames]);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
